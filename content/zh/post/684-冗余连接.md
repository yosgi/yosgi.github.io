# 684. 冗余连接

**发布日期：** 2021年07月07日

用时:15min

利用并查集模板即可

```javascript
var findRedundantConnection = function(edges) {
  var n = edges.length  var fa = new Array(n + 1)
  var find = function (x) {
    if (x != fa[x]) {
      fa[x] = find(fa[x])
    }
    return fa[x]
  }
  for(let i = 1 ; i <= n ; i ++) {
    fa[i] = i
  }
  for(let i = 0 ; i < n ; i ++) {
    var [x,y] = edges[i]
    if (find(x) === find(y)) {
      return [x,y]
    } else {
      fa[find(x)] = find(y)
    }
  }
};
```
