# 783. 二叉搜索树节点最小距离

**发布日期：** 2021年03月18日

用时 ： 10min

利用前序遍历二叉树递增的性质解决

```javascript
var minDiffInBST = function(root) {
    var min = Infinity    var pre = null    var dfs = function (root) {
        if (!root) return        if (root.left) {
            dfs(root.left)
        }
        if (pre == null) {
            pre = root.val        } else {
            console.log(pre,root.val)
            var reduce = Math.abs(pre - root.val)
            min = Math.min(min,reduce)
            pre = root.val        }
        if (root.right) {
            dfs(root.right)
        }
    }
    dfs(root)
    return min
};
```
